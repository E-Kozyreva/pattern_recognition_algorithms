class Autoencoder(nn.Module):
    def __init__(self):
        super(Autoencoder, self).__init__()
        self.encoder = nn.Sequential(
            nn.Linear(28*28, 128),
            nn.ReLU(),
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Linear(64, 32),
            nn.ReLU()
        )
        self.decoder = nn.Sequential(
            nn.Linear(32, 64),
            nn.ReLU(),
            nn.Linear(64, 128),
            nn.ReLU(),
            nn.Linear(128, 28*28),
            nn.Tanh()
        )

    def forward(self, x):
        x = self.encoder(x)
        x = self.decoder(x)
        return x

# Перенос модели на устройство
model = Autoencoder().to(device)

num_epochs = 10
for epoch in range(num_epochs):
    for data in train_loader:
        img, _ = data
        # Перенос данных на устройство
        img = img.view(img.size(0), -1).to(device)  
        output = model(img)
        loss = criterion(output, img)
        
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        
    print(f'Epoch [{epoch+1}/{num_epochs}], Loss: {loss.item():.4f}')

    idx1 = random.randint(0, len(test_dataset) - 1)
    idx2 = random.randint(0, len(test_dataset) - 1)
    
    img1, _ = test_dataset[idx1]
    img2, _ = test_dataset[idx2]
    
    # Перенос данных на устройство
    img1 = img1.to(device)  
    img2 = img2.to(device) 
    
    encoded_img1 = model.encoder(img1.view(-1, 28 * 28))
    encoded_img2 = model.encoder(img2.view(-1, 28 * 28))
    
    num_steps = 10
    interpolated = []
    for i in range(num_steps + 1):
        alpha = i / num_steps
        interpolated_encoded = alpha * encoded_img1 + (1 - alpha) * encoded_img2
        interpolated.append(interpolated_encoded)

    decoded_interpolated = [model.decoder(enc).view(-1, 1, 28, 28) for enc in interpolated]

    plt.figure(figsize=(20, 4))
    for i in range(num_steps + 1):
        ax = plt.subplot(1, num_steps + 1, i + 1)
         # Перенос изображения на CPU для отображения
        plt.imshow(decoded_interpolated[i].cpu().numpy()[0, 0], cmap='gray') 
        plt.title(f'Interpolation {i}')
        ax.axis('off')
    plt.show()

Вывод¶






















Матрица корреляции признаков скрытого представления данных демонстрирует степень взаимосвязи между признаками. Значение корреляции близкое к 1 указывает на сильную положительную корреляцию между признаками, тогда как значение близкое к -1 указывает на сильную отрицательную корреляцию. Значение близкое к 0 говорит о независимости признаков.


Анализ матрицы корреляции показывает, что максимально скоррелированные признаки имеют значения корреляции, не превышающие 0.6, что указывает на сильную положительную корреляцию между ними. Это может свидетельствовать о схожих характеристиках или аспектах данных, описываемых этими признаками. Негативно скоррелированных признаков обнаружено немного, что указывает на отсутствие явных противоречий в данных.


Таким образом, анализ матрицы корреляции позволяет понять взаимосвязь признаков в скрытом представлении данных автоэнкодера, что может быть полезно для дальнейшего анализа и интерпретации результатов модели.
